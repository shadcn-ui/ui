## Usage

```svelte showLineNumbers
<script lang="ts">
  import * as Pagination from "$lib/components/ui/pagination/index.js";
</script>
```

```svelte showLineNumbers
<Pagination.Root count={30} perPage={10} page={2}>
  {#snippet children({ pages, currentPage })}
    <Pagination.Content>
      <Pagination.Item>
        <Pagination.Previous />
      </Pagination.Item>
      {#each pages as page (page.key)}
        {#if page.type === "ellipsis"}
          <Pagination.Item>
            <Pagination.Ellipsis />
          </Pagination.Item>
        {:else}
          <Pagination.Item>
            <Pagination.Link {page} isActive={currentPage === page.value}>
              {page.value}
            </Pagination.Link>
          </Pagination.Item>
        {/if}
      {/each}
      <Pagination.Item>
        <Pagination.Next />
      </Pagination.Item>
    </Pagination.Content>
  {/snippet}
</Pagination.Root>
```

## Composition

Use the following composition to build a `Pagination`:

```text
Pagination.Root
└── Pagination.Content
    ├── Pagination.Item
    │   └── Pagination.Previous
    ├── Pagination.Item
    │   └── Pagination.Link
    ├── Pagination.Item
    │   └── Pagination.Ellipsis
    └── Pagination.Item
        └── Pagination.Next
```
