## Usage

```tsx showLineNumbers
import { Tooltip, TooltipTrigger } from "@/components/ui/tooltip"
```

```tsx showLineNumbers
<TooltipTrigger>
  <Button>Hover</Button>
  <Tooltip>
    <p>Add to library</p>
  </Tooltip>
</TooltipTrigger>
```

## Composition

Use the following composition to build a `Tooltip`:

```text
TooltipTrigger
├── Button
└── Tooltip
```
