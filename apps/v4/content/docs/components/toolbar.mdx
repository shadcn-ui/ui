## Usage

```tsx showLineNumbers
import {
  Toolbar,
  ToolbarButton,
  ToolbarGroup,
  ToolbarLink,
  ToolbarSeparator,
  ToolbarToggleGroup,
  ToolbarToggleItem,
} from "@/components/ui/toolbar"
```

```tsx showLineNumbers
<Toolbar aria-label="Formatting options">
  <ToolbarGroup>
    <ToolbarButton variant="outline">Bold</ToolbarButton>
    <ToolbarButton variant="outline">Italic</ToolbarButton>
  </ToolbarGroup>
  <ToolbarSeparator />
  <ToolbarGroup>
    <ToolbarButton variant="outline">Undo</ToolbarButton>
    <ToolbarButton variant="outline">Redo</ToolbarButton>
  </ToolbarGroup>
  <ToolbarSeparator />
  <TooolbarToggleGroup type="single" defaultValue="center">
    <ToolbarToggle value="start" variant="outline">Align Start</ToolbarToggle>
    <ToolbarToggle value="center" variant="outline">Align Center</ToolbarToggle>
    <ToolbarToggle value="end" variant="outline">Align End</ToolbarToggle>
  </TooolbarToggleGroup>
  <ToolbarSeparator />
  <ToolbarLink>
  	Edited 2 hours ago
  </ToolbarLink>
</Toolbar>
```
