## Usage

```gts showLineNumbers
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/ui/select';
```

```hbs showLineNumbers
<Select>
  <SelectTrigger @class="w-[180px]">
    <SelectValue @placeholder="Theme" />
  </SelectTrigger>
  <SelectContent>
    <SelectItem @value="light">Light</SelectItem>
    <SelectItem @value="dark">Dark</SelectItem>
    <SelectItem @value="system">System</SelectItem>
  </SelectContent>
</Select>
```
