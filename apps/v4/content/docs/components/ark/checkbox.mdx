## Checked State

Use `defaultChecked` for uncontrolled checkboxes, or `checked` and
`onCheckedChange` to control the state.

```tsx showLineNumbers
import * as React from "react"
import {
  Checkbox,
  CheckboxControl,
  CheckboxHiddenInput,
  CheckboxIndicator,
  type CheckboxCheckedChangeDetails,
} from "@/components/ui/checkbox"

export function Example() {
  const [checked, setChecked] = React.useState(false)

  return (
    <Checkbox
      checked={checked}
      onCheckedChange={(details: CheckboxCheckedChangeDetails) =>
        setChecked(!!details.checked)
      }
    >
      <CheckboxControl>
        <CheckboxIndicator />
      </CheckboxControl>
      <CheckboxHiddenInput />
    </Checkbox>
  )
}
```
