## Usage

```tsx showLineNumbers
import {
  InputOTP,
  InputOTPGroup,
  InputOTPSeparator,
  InputOTPSlot,
} from "@/components/ui/input-otp"
```

```tsx showLineNumbers
<InputOTP count={6}>
  <InputOTPGroup>
    <InputOTPSlot index={0} />
    <InputOTPSlot index={1} />
    <InputOTPSlot index={2} />
  </InputOTPGroup>
  <InputOTPSeparator />
  <InputOTPGroup>
    <InputOTPSlot index={3} />
    <InputOTPSlot index={4} />
    <InputOTPSlot index={5} />
  </InputOTPGroup>
</InputOTP>
```

## Pattern

Use the `type` prop to restrict input to digits only or alphanumeric.

```tsx showLineNumbers /type/
<InputOTP count={6} type="numeric">
  ...
</InputOTP>
```

```tsx showLineNumbers /type/
<InputOTP count={6} type="alphanumeric">
  ...
</InputOTP>
```

### Controlled

Use the `value` and `onValueChange` props to control the input value.

```tsx showLineNumbers
const [value, setValue] = React.useState<string[]>([])

<InputOTP count={6} value={value} onValueChange={(details) => setValue(details.value)}>
  ...
</InputOTP>
```
