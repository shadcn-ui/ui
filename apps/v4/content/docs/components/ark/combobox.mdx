## Usage

```tsx showLineNumbers
import {
  Combobox,
  ComboboxClearTrigger,
  ComboboxContent,
  ComboboxControl,
  ComboboxInput,
  ComboboxItem,
  ComboboxItemText,
  ComboboxList,
  ComboboxTrigger,
  useFilter,
  useListCollection,
} from "@/components/ui/combobox"
```

```tsx showLineNumbers
const frameworkItems = [
  { label: "Next.js", value: "nextjs" },
  { label: "SvelteKit", value: "sveltekit" },
  { label: "Nuxt.js", value: "nuxtjs" },
  { label: "Remix", value: "remix" },
  { label: "Astro", value: "astro" },
]

export function ExampleCombobox() {
  const { contains } = useFilter({ sensitivity: "base" })
  const { collection, filter } = useListCollection({
    initialItems: frameworkItems,
    filter: contains,
  })

  return (
    <Combobox
      collection={collection}
      onInputValueChange={(details) => filter(details.inputValue)}
    >
      <ComboboxControl>
        <ComboboxInput placeholder="Select a framework" />
        <ComboboxClearTrigger />
        <ComboboxTrigger />
      </ComboboxControl>
      <ComboboxContent>
        <ComboboxList>
          {collection.items.map((item) => (
            <ComboboxItem key={item.value} item={item}>
              <ComboboxItemText>{item.label}</ComboboxItemText>
            </ComboboxItem>
          ))}
        </ComboboxList>
      </ComboboxContent>
    </Combobox>
  )
}
```

## Custom Items

Use `useListCollection` with custom item objects. The `label` field is used for filtering and display.

```tsx showLineNumbers
import {
  Combobox,
  ComboboxClearTrigger,
  ComboboxContent,
  ComboboxControl,
  ComboboxInput,
  ComboboxItem,
  ComboboxItemText,
  ComboboxList,
  ComboboxTrigger,
  useFilter,
  useListCollection,
} from "@/components/ui/combobox"

const countryItems = [
  { label: "Argentina", value: "argentina", continent: "South America" },
  { label: "Australia", value: "australia", continent: "Oceania" },
  { label: "Brazil", value: "brazil", continent: "South America" },
]

export function ExampleComboboxCustomItems() {
  const { contains } = useFilter({ sensitivity: "base" })
  const { collection, filter } = useListCollection({
    initialItems: countryItems,
    filter: contains,
  })

  return (
    <Combobox
      collection={collection}
      onInputValueChange={(details) => filter(details.inputValue)}
    >
      <ComboboxControl>
        <ComboboxInput placeholder="Search countries..." />
        <ComboboxClearTrigger />
        <ComboboxTrigger />
      </ComboboxControl>
      <ComboboxContent>
        <ComboboxList>
          {collection.items.map((item) => (
            <ComboboxItem key={item.value} item={item}>
              <ComboboxItemText>{item.label}</ComboboxItemText>
            </ComboboxItem>
          ))}
        </ComboboxList>
      </ComboboxContent>
    </Combobox>
  )
}
```

## Multiple Selection

Use the `multiple` prop for multi-select behavior. Use `ComboboxContext` to access `selectedItems` and render tags with `ComboboxTagsControl`, `ComboboxTag`, and `ComboboxTagsInput`.

```tsx showLineNumbers
import {
  Combobox,
  ComboboxContent,
  ComboboxContext,
  ComboboxItem,
  ComboboxItemText,
  ComboboxList,
  ComboboxTag,
  ComboboxTagsControl,
  ComboboxTagsInput,
  ComboboxTrigger,
  useFilter,
  useListCollection,
} from "@/components/ui/combobox"

const frameworkItems = [
  { label: "Next.js", value: "nextjs" },
  { label: "SvelteKit", value: "sveltekit" },
  { label: "Nuxt.js", value: "nuxtjs" },
  { label: "Remix", value: "remix" },
  { label: "Astro", value: "astro" },
]

export function ExampleComboboxMultiple() {
  const { contains } = useFilter({ sensitivity: "base" })
  const { collection, filter } = useListCollection({
    initialItems: frameworkItems,
    filter: contains,
  })

  return (
    <Combobox
      collection={collection}
      onInputValueChange={(details) => filter(details.inputValue)}
      multiple
      defaultValue={["nextjs"]}
    >
      <ComboboxContext>
        {(context) => (
          <ComboboxTagsControl>
            {context.selectedItems.map((item) => (
              <ComboboxTag
                key={item.value}
                value={item.value}
                onRemove={(value) => context.clearValue(value)}
              >
                {item.label}
              </ComboboxTag>
            ))}
            <ComboboxTagsInput placeholder="Select frameworks" />
            <ComboboxTrigger />
          </ComboboxTagsControl>
        )}
      </ComboboxContext>
      <ComboboxContent>
        <ComboboxList>
          {collection.items.map((item) => (
            <ComboboxItem key={item.value} item={item}>
              <ComboboxItemText>{item.label}</ComboboxItemText>
            </ComboboxItem>
          ))}
        </ComboboxList>
      </ComboboxContent>
    </Combobox>
  )
}
```
