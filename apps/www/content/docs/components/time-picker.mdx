## Usage

```typescript jsx
import {useState} from "react";
const [time,setTime]=useState(0)
return <TimePickerContainer onTimeChange={(v)=>setTime(v)}>
    <TimePicker
        timeMilliseconds={1}
        minValue={0}
        maxValue={1000}
        prefixLabel={<span>Milliseconds:</span>}
        suffixLabel={<span>ms</span>}
        step={1}
        onValueChange={(value)=>console.log("Time value change to : "+value)}
    />
</TimePickerContainer >
```
