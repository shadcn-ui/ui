## Usage

```tsx
import {
  Editable,
  EditableArea,
  EditableCancel,
  EditableInput,
  EditableLabel,
  EditablePreview,
  EditableSubmit,
  EditableToolbar,
} from "@/components/ui/editable"
```

```tsx
<Editable>
  <EditableLabel>Fruit</EditableLabel>
  <EditableArea>
    <EditablePreview />
    <EditableInput placeholder="Apple" />
    <EditableTrigger>Edit</EditableTrigger>
  </EditableArea>
  <EditableToolbar>
    <EditableCancel>Cancel</EditableCancel>
    <EditableSubmit>Save</EditableSubmit>
  </EditableToolbar>
</Editable>
```
