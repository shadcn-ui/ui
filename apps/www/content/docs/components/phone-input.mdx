## Usage

```tsx
import { PhoneInput, type PhoneInputValue } from "@/components/ui/phone-input"
```

```tsx
const [value, setValue] = React.useState<PhoneInputValue>()

return <PhoneInput value={value} onChange={setValue} />
```
